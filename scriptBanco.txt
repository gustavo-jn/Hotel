CREATE DATABASE hotelaria CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;
USE hotelaria;


CREATE TABLE hero(
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    subtitulo VARCHAR(255) NOT NULL,
    imagem VARCHAR(255) NOT NULL
);

CREATE TABLE sobre (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(255) DEFAULT 'Sobre Nós',
    texto TEXT NOT NULL,
    imagem VARCHAR(255) NOT NULL,
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    atualizado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE equipe (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cargo VARCHAR(100) NOT NULL,
    facebook VARCHAR(255),
    twitter VARCHAR(255),
    linkedin VARCHAR(255),
    imagem VARCHAR(255) NOT NULL
);

CREATE TABLE faq (
    id_faq INT AUTO_INCREMENT PRIMARY KEY,
    pergunta VARCHAR(255) NOT NULL,
    resposta TEXT NOT NULL,
    imagem VARCHAR(255) DEFAULT NULL,
    ordem INT DEFAULT 0,
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE contato (
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) DEFAULT NULL,
    email VARCHAR(150) DEFAULT NULL,
    mensagem TEXT DEFAULT NULL,
    data_envio TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE imagem(
    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    imagem VARCHAR(255) DEFAULT NULL,
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);



-- Tabela de usuários
CREATE TABLE usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    senha VARCHAR(255) NOT NULL,
    telefone VARCHAR(20),
    tipo ENUM('user','admin') DEFAULT 'user',
    foto VARCHAR(255) DEFAULT 'default.jpg',
    criado_em DATETIME DEFAULT CURRENT_TIMESTAMP
);


-- Tabela de quartos
CREATE TABLE quartos (
    id_quarto INT AUTO_INCREMENT PRIMARY KEY,
    numero_quarto VARCHAR(10) NOT NULL,
    tipo_quarto VARCHAR(50) NOT NULL,
    preco_diaria DECIMAL(10,2) NOT NULL,
    descricao TEXT,
    imagem VARCHAR(255),
    status ENUM('disponivel','ocupado','manutencao') DEFAULT 'disponivel'
);

-- Tabela de reservas
CREATE TABLE reservas (
    id_reserva INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    id_quarto INT NOT NULL,
    data_checkin DATE NOT NULL,
    data_checkout DATE NOT NULL,
    status ENUM('pendente','confirmada','cancelada') DEFAULT 'pendente',
    criada_em DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_quarto) REFERENCES quartos(id_quarto) ON DELETE CASCADE
);
